1. Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(*) as `n. iscritti`, YEAR(`enrolment_date`) as `anno iscrizione` FROM `students` GROUP BY YEAR(`enrolment_date`);

2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT COUNT(*) as `n. insegnanti`, `office_address` as `ufficio` FROM `teachers` GROUP BY `office_address`;
